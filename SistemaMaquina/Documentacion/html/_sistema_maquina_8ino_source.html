305* NAME: mostrarTemp
306*----------------------------------------------------------------------------
307* PARAMS: none
308* return: none
309*----------------------------------------------------------------------------
310* PURPOSE:
311* Reads humidity and temperature data from a DHT11 sensor and displays it on both the Serial monitor and an LCD display
312*----------------------------------------------------------------------------
313* NOTE:
314*
315*****************************************************************************/
316void mostrarTemp(){
317 Serial.println("nivel de humedad - temperatura");
318 Serial.print("DHT11, \t");
319 int chk = DHT.read11(DHT11_PIN);
320 switch (chk)
321 {
322 case DHTLIB_OK:
323 Serial.print("OK,\t");
324 break;
325 case DHTLIB_ERROR_CHECKSUM:
326 Serial.print("Checksum error,\t");
327 break;
328 case DHTLIB_ERROR_TIMEOUT:
329 Serial.print("Time out error,\t");
330 break;
331 default:
332 Serial.print("Unknown error,\t");
333 break;
334 }
335 Serial.print(DHT.getHumidity(), 1);
336 lcd.setCursor(0, 0);
337 lcd.print("H");
338 lcd.setCursor(2, 0);
339 lcd.print(DHT.getHumidity(), 1);
340 lcd.setCursor(0, 1);
341 lcd.print("T");
342 lcd.setCursor(2, 1);
343 Serial.print(",\t");
344 tempValue = DHT.getTemperature();
345 Serial.println(tempValue, 1);
346 lcd.print(tempValue, 1);
347}
